import numpy as np
import matplotlib.pyplot as plt

lambdas = [1, 10, 100, 1000]
n = 100

beta_values = np.logspace(0, 3, 100)
alpha_values = [1, 10, 100, 1000]

for l in lambdas:
    plt.figure()
    for a in alpha_values:
        mse_values = []
        for b in beta_values:
            # MSE from P1 iii
            mse = (n * l + (a - l * b) ** 2) / (n + b) ** 2
            mse_values.append(mse)
        plt.plot(beta_values, mse_values, label=f'={a}')
    plt.xscale('log')
    plt.yscale('log')
    plt.xlabel('')
    plt.ylabel('MSE')
    plt.title(f'MSE of Estimator for ={l}')
    plt.legend()
    plt.grid(True)
    plt.show()

ii

For lower $\lambda$ values, there are clear local minima for the MSE of given certain $\alpha, \beta$ values. However, as $\lambda$ increases, the MSE also increases exponenentialy with no local minima apart from larger and larger $\alpha$ values as $\beta$ also increases.

iii

In Problem 1, we concluded that the estimator is unbiased when $\alpha = \lambda\beta$. We can check this by identifying the values of $\alpha, \beta, \lambda$ at the minima in the graphs and find that they do agree.

l = 1
N = 1000

ab_values = [(1, 1), (1, 10), (10, 1), (10, 10)]
n = 10

lambdahat_values = []

np.random.seed(0)
for a, b in ab_values:
    temp = []
    for _ in range(N):
        samples = np.random.poisson(l, n)
        lambdahat = (np.sum(samples) + a) / (n + b)
        temp.append(lambdahat)
    lambdahat_values.append(temp)

def expected_mean(a, b, n, l):
    return (n * l + a) / (n + b)

def expected_variance(a, b, n, l):
